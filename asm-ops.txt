

eax contains the data pointer

"." - Output byte
mov     rax, 1
mov     rdx, 1
mov     rsi, PTR
mov     rdi, 1
syscall

"," - Read byte
mov rax, 0
mov rdx, 1
mov rsi, PTR
mov rdi, 0
syscall


">" - Increment pointer
inc     PTR

{ 0x40 }

"<" - Decrement pointer
dec     PTR

{ 0x48 }

"+" - Increment data
inc QWORD PTR [PTR]


"-" - Decrement data
dec     QWORD PTR [PTR]


"[" - Loop start
LOOP_START:
cmp     [eax],  0     ; 0x83, 0x38, 0x00
je      $LOOP_END     ; 0x74, {number of skipped bytes}

"]" - Loop end
jmp     {number of bytes to go back+2, negative in two's complement}
